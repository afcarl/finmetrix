def kosenok_opt(f, x, options):
    """ Optimizer function

    Arguments:
        f : function object which depends on x (float)
        x : initial value of the parameter (2 vector)
        options : things like tolerance, max iterations, and initial delta (dictionary)
    
    Returns:
        f(x) : function value at the last iteration (float)
        x : parameter value at the last iteration (2 vector)
        pathx : array of stored iterations of x (i x 2 array)
        pathy : array of stored iterations of f(x) (i vector)
    
    """
    # Initialize
    delta = options['delta']
    # Iteration counter
    i = 0
    # Arrays for storing iterations of x and f(x)
    pathx, pathy = [x], [f(x)]
    
    # Repeat until
    # - either delta is too small
    # - or number of iterations is too big
    while (delta > options['tol']) and (i < options['maxiter']):
        
        # Store current delta for the next iteration
        delta0 = delta
        
        # For each element in the parameter vector
        for p in range(len(x)):
            
            # Do while something does not break the loop
            while True:
                
                # Create copies of current x
                x1, x2 = x.copy(), x.copy()
                # Step to the left and to the right from the current coordinate
                x1[p] += delta
                x2[p] -= delta
                # Evaluate function at each point
                f0, f1, f2 = f(x), f(x1), f(x2)
                
                # If the function has decreased,
                if f1 - f0 < 0:
                    # ... update current x,
                    x = x1
                    # ... and increase the speed of descend
                    delta *= 2
                # Same condition for the opposite step
                elif f2 - f0 < 0:
                    x = x2
                    delta *= 2
                # If we fail to descend, then go to the next coordinate by breaking the loop
                else:
                    break
                
                # Append history of iterations
                pathx.append(x)
                pathy.append(f(x))
        
        # Increment iteration counter
        i += 1
        # Slow down descend
        delta = delta0 / 2
        
        # Print iteration info
        print 'Iter:', i, ', x[i] =', x, ', Delta = %.1e' % delta
    
    return f(x), x, pathx, pathy

Define Rosenbrock function¶

In [3]:
# Rosenbrock function
def rosen(x):
    return 100 * (x[1] - x[0]**2)**2 + (1 - x[0])**2
