Fill empty values¶

In [3]: 
def fg(group):
    y = np.array(group.rate)
    x = np.array(group.days)
    a = group.days.min()
    b = group.days.max()
    new_x = np.linspace(a, b, b - a + 1).astype(int)
    new_y = interp1d(x, y, kind = 'cubic')(new_x)
    
    group = ps.DataFrame({'rate' : new_y}, index = new_x)
    return group

new_rates = yields.reset_index().groupby('date').apply(fg)
new_rates = new_rates['rate'].unstack().fillna(method = 'ffill', axis = 1).fillna(method = 'bfill', axis = 1)
